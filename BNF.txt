<Program> -> <FDefns><Main>
<FDefns> -> Declaration | Lambda                 //declarations and block of text
<Main> -> "LOGIN" <Block> "LOGOUT"
<Block> -> <Exp><Newline> | <Exp><Newline><Block>
<Exp> -> <Loop> | <If> | <Assignment> | <Call> | "UNFOLLOW" | "BLOCK" | "FOLLOW" | <Return>
<Return> ->  "REPORT" <INT> |"REPORT" <CHAR> |"REPORT" <FLOAT> |"REPORT" <STRING> |"REPORT" <BOOL> | "REPORT" <Varname>
<If> ->"IF(" <Boolean> ")follow{" <Block> "}" <Elseif><Else>
<Elseif> -> <Elseif> "ELSEIF(" <Boolean> ")follow{" <Block> "}"| Lambda
<Else> -> "ELSE(" <Boolean> ")follow{" <Block> "}" | Lambda
<Loop> -> "RT(" <Boolean> "){" <Block> "}"
<Assignment> -> <DType> <Varname> "=" <Varname>
                |"@INT" <Varname> "=" <INT>
                | "@CHIRP" <Varname> "=" <CHAR>
                | "@COKE" <Varname> "=" <FLOAT>
                | "@MSG" <Varname> "=" <STRING>
                | "@TRALSE" <Varname> "=" <BOOL>
<Dtype> = "@INT" | "@CHIRP" | "@COKE" | "@MSG" | "@TRALSE"
<Newline> -> "\n"<Newline> | "\n"
<Boolean> -> <VName> ">=" <Vname> | <Vname> "<=" <Vname> | <Vname> "==" <Vname> | <Vname> ">" <Vname> | <Vname> "<" <Vname>
<INT>
<CHAR>
<FLOAT>
<STRING>
<BOOL>
<Declaration> -> <Dtype> <Varname> "(" <Args> "){" <Block> "}"
<Call> -> <Varname> "(" <Args> ")"
<Args> -> <Dtype> <varname> | <Dtype> <varname>  ", " <Args> | Lambda
