<Main> -> "login" <Block> "logout"
<Block> -> <Exp><Newline> | <Exp><Newline><Block>
<Exp> -> <Loop> | <If> | <Assignment> | <Call> | "UNFOLLOW" | "BLOCK" | "FOLLOW"
<If> ->"IF(" <Boolean> "){" <Block> "}"
<Loop> -> "RT(" <Boolean> "){" <Block> "}"
<Assignment> -> "@INT" <Varname> "=" <INT> | "@CHIRP" <Varname> "=" <CHAR> | "@COKE" <Varname> "=" <FLOAT> | "@MSG" <Varname> "=" <STRING> | "@TRALSE" <Varname> "=" <BOOL>
<Newline> -> "\n"
<Boolean> -> <VName> ">=" <Vname> | <Vname> "<=" <Vname> | <Vname> "==" <Vname> | <Vname> ">" <Vname> | <Vname> "<" <Vname>
