<Program> -> <FDefns><Main>
<FDefns> -> NULL |                  //declarations and block of text
<Main> -> "login" <Block> "logout"
<Block> -> <Exp><Newline> | <Exp><Newline><Block>
<Exp> -> <Loop> | <If> | <Assignment> | <Call> | "UNFOLLOW" | "BLOCK" | "FOLLOW" | <return>
<return> ->  "REPORT" <INT> |"REPORT" <CHAR> |"REPORT" <FLOAT> |"REPORT" <STRING> |"REPORT" <BOOL>
<If> ->"IF(" <Boolean> "){" <Block> "}"
<Loop> -> "RT(" <Boolean> "){" <Block> "}"
<Dtype> = "@INT" | "@CHIRP" | "@COKE" | "@MSG" | "@TRALSE"
<Assignment> -> <DType> <Varname> "=" <Varname>
                |"@INT" <Varname> "=" <INT>
                | "@CHIRP" <Varname> "=" <CHAR>
                | "@COKE" <Varname> "=" <FLOAT>
                | "@MSG" <Varname> "=" <STRING>
                | "@TRALSE" <Varname> "=" <BOOL>
<Declaration> -> <Dtype> <Fname> "(" <args> "){" <Block> "}"
<Call> -> <Fname> "(" <Args> ")"
<Args> -> <Dtype> <varname> | <Dtype> <varname>  ", " <Args> | Lambda
<Newline> -> "\n"<Newline> | "\n"
<Boolean> -> <VName> ">=" <Vname> | <Vname> "<=" <Vname> | <Vname> "==" <Vname> | <Vname> ">" <Vname> | <Vname> "<" <Vname>
